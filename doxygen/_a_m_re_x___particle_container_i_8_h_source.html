918 template <int NStructReal, int NStructInt, int NArrayReal, int NArrayInt>
919 void
920 ParticleContainer<NStructReal, NStructInt, NArrayReal, NArrayInt>::CreateVirtualParticles (int level,
921  AoS& virts) const
922 {
923  BL_PROFILE("ParticleContainer<NSR, NSI, NAR, NAI>::CreateVirtualParticles()");
924  BL_ASSERT(level > 0);
925  BL_ASSERT(virts.empty());
926 
927  if (level >= int(m_particles.size()))
928  //
929  // This level could exist and simply have no particles.
930  //
931  return;
932  //
933  // Read these from the parm file if we haven't done so yet.
934  //
935  if (aggregation_type == "")
936  {
937  ParmParse pp("particles");
938  aggregation_type = "None";
939  pp.query("aggregation_type",aggregation_type);
940  aggregation_buffer = 2;
941  pp.query("aggregation_buffer",aggregation_buffer);
942  }
943  //
944  // Create a buffer so that particles near the cf border are not aggregated.
945  //
946  BoxArray buffer = amrex::complementIn(Geom(level).Domain(), ParticleBoxArray(level));
947 
948  buffer.grow(aggregation_buffer);
949 
950  const auto& pmap = m_particles[level];
951  for (const auto& kv : pmap) {
952  const auto& pbox = kv.second.GetArrayOfStructs();
953 
954  //
955  // Map for use in Cell aggregation.
956  //
957  std::map<IntVect,ParticleType> agg_map;
958 
959  for (auto it = pbox.cbegin(); it != pbox.cend(); ++it)
960  {
961  IntVect cell = Index(*it, level);
962  if (buffer.contains(cell))
963  {
964  // It's in the no-aggregation buffer.
965  // Set its id to indicate that it's a virt.
966  virts.push_back(*it);
967  virts.back().m_idata.id = VirtualParticleID;
968  }
969  else
970  {
971  if (aggregation_type == "None")
972  {
973  //
974  // No aggregation. Simply clone the particle.
975  // Set its id to indicate that it's a virt.
976  virts.push_back(*it);
977  virts.back().m_idata.id = VirtualParticleID;
978  }
979  else if (aggregation_type == "Cell")
980  {
981  //
982  // Note that Cell aggregation assumes that p.m_rdata.arr[BL_SPACEDIM] is mass and
983  // that all other components should be combined in a mass-weighted
984  // average.
985  //
986  auto agg_map_it = agg_map.find(cell);
987 
988  if (agg_map_it == agg_map.end())
989  {
990  //
991  // Add the particle.
992  //
993  ParticleType p = *it;
994  //
995  // Set its id to indicate that it's a virt.
996  //
997  p.m_idata.id = VirtualParticleID;
998  agg_map[cell] = p;
999  }
1000  else
1001  {
1002  BL_ASSERT(agg_map_it != agg_map.end());
1003  const ParticleType& pnew = *it;
1004  ParticleType& pold = agg_map_it->second;
1005  const Real old_mass = pold.m_rdata.arr[BL_SPACEDIM];
1006  const Real new_mass = pnew.m_rdata.arr[BL_SPACEDIM];
1007  const Real total_mass = old_mass + new_mass;
1008  //
1009  // Set the position to the center of mass.
1010  //
1011  for (int i = 0; i < BL_SPACEDIM; i++)
1012  {
1013  pold.m_rdata.pos[i] = (old_mass*pold.m_rdata.pos[i] + new_mass*pnew.m_rdata.pos[i])/total_mass;
1014  }
1015  BL_ASSERT(this->Index(pold, level) == cell);
1016  //
1017  // Set the metadata (presumably velocity) to the mass-weighted average.
1018  //
1019  for (int i = BL_SPACEDIM + 1; i < BL_SPACEDIM + NStructReal; i++)
1020  {
1021  pold.m_rdata.arr[i] = (old_mass*pold.m_rdata.arr[i] + new_mass*pnew.m_rdata.arr[i])/total_mass;
1022  }
1023  pold.m_rdata.arr[BL_SPACEDIM] = total_mass;
1024  }
1025  }
1026  else if (aggregation_type == "Flow")
1027  {
1028  amrex::Abort("Flow aggregation not implemented");
1029  }
1030  else
1031  {
1032  amrex::Abort("Unknown Particle Aggregation mode");
1033  }
1034  }
1035  }
1036  if (aggregation_type == "Cell")
1037  {
1038  //
1039  // Add the aggregated particles to the virtuals.
1040  //
1041  for (const auto& agg_particle : agg_map)
1042  {
1043  virts.push_back(agg_particle.second);
1044  }
1045  }
1046  }
1047 }
