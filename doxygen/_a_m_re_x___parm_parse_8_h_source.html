16 //
17 // ParmParse class implements a simple database for the storage and
18 // retrieval of command-line and input-file arguments. The entries are
19 // stored in a static table in (name,value_list) pairs.
20 //
21 // The format of the input file is a series of DEFINITIONS.
22 //
23 // A DEFINITION is of the form <name> = <value> <value> ...
24 // The equal sign is important since the list of values can span multiple
25 // lines.
26 //
27 // Comments in an input file include all text from a '#' character to the
28 // end of the line. Here is an example input file:
29 //
30 // niter = 100 # niter is an integer
31 // title = "Double Wammy" # example of a string with spaces
32 // cell_size = 0.5 0.75 # cell spacing in each dimension
33 // plot.var = Density 1 10 # a list of values
34 // plot.var = Energy 5 12 # another list of values
35 // bigarray = 1 2 3 4 5 6 7 8 # first part of array
36 // 9 10 11 12 # continuation of bigarray
37 // test = apple "boy blue" 10 20 30 40
38 // FILE = prob_file # insert contents of this "prob_file" here
39 //
40 // The "FILE = <filename>" definition is special. Rather than just
41 // adding this entry to the database, it reads the contents of <filename>
42 // into the database.
43 //
44 // ParmParse stores all entries in a static table which is built the
45 // first time a ParmParse object is constructed (usually in main()).
46 // Subsequent invocations have access to this table.
47 // A ParmParse constructor has an optional "prefix" argument that will
48 // limit the searches to only those entries of the table with this prefix
49 // in name. For example:
50 // ParmParse pp("plot");
51 // will find only those entries with name given by "plot.<string>".
52 //
53 // All values in the table are stored as strings. For example, the
54 // values of "cell_size" in the above input file are stored as the
55 // strings "0.5" and "0.75". These strings can be returned as either
56 // strings or numeric values by the query functions.
57 // Character strings with spaces must be delimited by double quotes
58 // in the input file but the quotes are stripped before they are entered
59 // into the table. For example, 'title' in the above input file has a
60 // single value, the string 'Double Wammy' (without the quotes).
61 // Each value in the list associated with a definition can be referred to
62 // by its index number. The index numbers start at 0 just like an array
63 // in the C programming language. Consider the definition of "test" in
64 // the above input file. The first value 'apple'is a string with index
65 // 0. The second value 'boy blue' is a string with index 1. The
66 // remaining four values are integers indexed 2, 3, 4, and 5.
67 //
68 // For a string value to represent an integer or float it must fit the
69 // following regular experssion:
70 // Sign ::= '+' | '-'
71 // Digit ::= '0' | '1' | ... | '9'
72 // Integer ::= [Sign]Digit+
73 // Exp ::= ('e'|'E')Integer
74 // Float ::= ( Integer[.Digit*][Exp] | [Integer].Digit+[Exp] )
75 //
76 // Where '+' indicates one or more occurences, '*' represents zero or
77 // more occurences, '|' means one or the other and '[]' represents zero
78 // or one occurence.
79 //
80 // Note that floats and doubles have the same string representation and
81 // that the FORTRAN "double" exponent format is not supported.
82 // That is, 1.0d+3 is not a valid representation of a floating point
83 // number but that 1.0e+3 is acceptable.
84 //
85 // There are a host of functions allowing the user to query the database
86 // and retrieve values. Here are some general rules about the names of
87 // the member functions:
88 //
89 // * Functions with the string "get" in their names attempt to get a
90 // value or an array of values from the table. They generate a
91 // run-time error if they are not successful.
92 //
93 // * Functions with the string "query" in their names attempt to get a
94 // value or an array of values from the table. They return the value 1
95 // (true) if they are successful and 0 (false) if not.
96 //
97 // * Functions with the string "kth" in their names get values from
98 // the Kth entry with the given name in the database. This is
99 // necessary since there may be multiple definitions with the same
100 // name in the database.
101 //
102 // * Functions without the string "kth" in their names get values from
103 // the last entry with the given name in the database. Note that the
104 // definitions from the command line are appended to the database table
105 // and hence will be the last entries.
106 //
107 // * Functions with the string "arr" in their names get an Array of
108 // values from the given entry in the table. The array argument is
109 // resized (if necessary) to hold all the values requested.
110 //
111 // * Functions without the string "arr" in their names get single
112 // values from the given entry in the table.
113 //
114 // The following is a code sample showing how to use ParmParse:
115 //
116 // main(int argc, char **argv)
117 // {
118 // char* in_file_name = argv[1];
119 // ParmParse::Initialize(argc-2, argv+2, in_file_name);
120 //
121 // // Query table for value of "niter". If not in table
122 // // then set to default value
123 // if (!pp.query("niter",niter)) niter = 20;
124 //
125 // // read array of cell sizes if in table
126 // Vector<float> dx;
127 // if (nx=pp.countval("cell_size")) {
128 // // get nx values starting at index 0 and store in dx.
129 // // dx is automatically resized here.
130 // pp.getarr("cell_size",dx,0,nx);
131 // }
132 // ParmParse::Finalize();
133 // }
134 //
135 // void do_graphics()
136 // {
137 // //
138 // // Will only query entries with the "plot" prefix:
139 // //
140 // ParmParse pp("plot");
141 // //
142 // // Read all variables with "plot.var" keyword.
143 // //
144 // std::string var_name;
145 // Vector<int> range;
146 // int num = pp.countname("var");
147 // for (int k = 0; k < num; k++)
148 // {
149 // //
150 // // Element 0 in list is a string.
151 // //
152 // pp.getkth("var",k,var_name,0);
153 // //
154 // // Elements 1 and 2 are integers.
155 // // Note that "range" will be resized to hold 2 elements.
156 // //
157 // pp.getktharr("var",k,range,1,2);
158 // cout << "variable = " << var_name << "lo, hi = ",
159 // << range[0] << " " << range[1] << endl;
160 // }
161 // }
162 // -----------------------------------------------------------------
163 // ----------------------- END COMMENTS ---------------------------
164 // -----------------------------------------------------------------

Gets an std::vector<int> of num_val values from kth occurrence of given name. If successful, the values are converted to an int and stored in the std::vector<int> object ref. ref is resized (if necessary) to hold num_val values.
