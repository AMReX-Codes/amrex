311 {
312  int aT_Size(-2);
313  if(myLocalId == rootId) {
314  aT_Size = aT.size();
315  }
316  ParallelDescriptor::Bcast(&aT_Size, 1, rootId, localComm);
317  BL_ASSERT(aT_Size >= 0);
318  if(myLocalId != rootId) {
319  aT.resize(aT_Size);
320  }
321  if(aT_Size > 0) {
322  ParallelDescriptor::Bcast(aT.dataPtr(), aT.size(), rootId, localComm);
