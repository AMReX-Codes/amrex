185  return M;
186  }
187  //
188  // The first element in fracs and cells is the lowest corner, the last is the highest.
189  //
190  const Real hilen[BL_SPACEDIM] = { AMREX_D_DECL((p.m_rdata.pos[0]-plo[0]+dx_part[0]/2)/dx_geom[0],
191  (p.m_rdata.pos[1]-plo[1]+dx_part[1]/2)/dx_geom[1],
192  (p.m_rdata.pos[2]-plo[2]+dx_part[2]/2)/dx_geom[2]) };
193

195  (p.m_rdata.pos[1]-plo[1]-dx_part[1]/2)/dx_geom[1],
196  (p.m_rdata.pos[2]-plo[2]-dx_part[2]/2)/dx_geom[2]) };
197 
198  const IntVect hicell(AMREX_D_DECL(floor(hilen[0]), floor(hilen[1]), floor(hilen[2])));
199 
200  const IntVect locell(AMREX_D_DECL(floor(lolen[0]), floor(lolen[1]), floor(lolen[2])));
201 
202  const Real cell_density = AMREX_D_TERM(dx_geom[0]/dx_part[0],*dx_geom[1]/dx_part[1],*dx_geom[2]/dx_part[2]);
203 
204  const int M = AMREX_D_TERM((hicell[0]-locell[0]+1),*(hicell[1]-locell[1]+1),*(hicell[2]-locell[2]+1));
205 
206  fracs.resize(M);
207  cells.resize(M);
208  //
209  // This portion might be slightly inefficient. Feel free to redo it if need be.
210  //
211  int i = 0;
212 #if (BL_SPACEDIM == 1)
213  for (int xi = locell[0]; xi <= hicell[0]; xi++)
214  {
215  cells[i][0] = xi;
216  fracs[i] = (std::min(hilen[0]-xi,Real(1))-std::max(lolen[0]-xi,Real(0)))*cell_density;
217  i++;
218  }
219 #elif (BL_SPACEDIM == 2)
220  for (int yi = locell[1]; yi <= hicell[1]; yi++)
221  {
222  const Real yf = std::min(hilen[1]-yi,Real(1))-std::max(lolen[1]-yi,Real(0));
223  for (int xi = locell[0]; xi <= hicell[0]; xi ++)
224  {
225  cells[i][0] = xi;
226  cells[i][1] = yi;
227  fracs[i] = yf * (std::min(hilen[0]-xi,Real(1))-std::max(lolen[0]-xi,Real(0)))*cell_density;
228  i++;
229  }
230  }
231 #elif (BL_SPACEDIM == 3)
232  for (int zi = locell[2]; zi <= hicell[2]; zi++)
233  {
234  const Real zf = std::min(hilen[2]-zi,Real(1))-std::max(lolen[2]-zi,Real(0));
235  for (int yi = locell[1]; yi <= hicell[1]; yi++)
236  {
237  const Real yf = std::min(hilen[1]-yi,Real(1))-std::max(lolen[1]-yi,Real(0));
238  for (int xi = locell[0]; xi <= hicell[0]; xi++)
239  {
240  cells[i][0] = xi;
241  cells[i][1] = yi;
242  cells[i][2] = zi;
243  fracs[i] = zf * yf * (std::min(hilen[0]-xi,Real(1))-std::max(lolen[0]-xi,Real(0))) * cell_density;
244  i++;
245  }
246  }
247  }
248 #endif
