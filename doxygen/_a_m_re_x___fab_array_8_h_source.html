53 /*
54  A Collection of Fortran Array-like Objects
55 
56 
57  The FabArray<FAB> class implements a collection (stored as an array) of
58  Fortran array-like objects. The parameterized type FAB is intended to be
59  any class derived from BaseFab<T>. For example, FAB may be a BaseFab of
60  integers, so we could write:
61 
62  FabArray<BaseFab<int> > int_fabs;
63 
64  Then int_fabs is a FabArray that can hold a collection of BaseFab<int>
65  objects.
66 
67  FabArray is not just a general container class for Fortran arrays. It is
68  intended to hold "grid" data for use in finite difference calculations in
69  which the data is defined on a union of (usually disjoint) rectangular
70  regions embedded in a uniform index space. This region, called the valid
71  region, is represented by a BoxArray. For the purposes of this discussion,
72  the Kth Box in the BoxArray represents the interior region of the Kth grid.
73 
74  Since the intent is to be used with finite difference calculations a
75  FabArray also includes the notion of a boundary region for each grid. The
76  boundary region is specified by the ngrow parameter which tells the FabArray
77  to allocate each FAB to be ngrow cells larger in all directions than the
78  underlying Box. The larger region covered by the union of all the FABs is
79  called the region of definition. The underlying notion is that the valid
80  region contains the grid interior data and the region of definition includes
81  the interior region plus the boundary areas.
82 
83  Operations are available to copy data from the valid regions into these
84  boundary areas where the two overlap. The number of components, that is,
85  the number of values that can be stored in each cell of a FAB, is either
86  given as an argument to the constructor or is inherent in the definition of
87  the underlying FAB. Each FAB in the FabArray will have the same number of
88  components.
89 
90  In summary, a FabArray is an array of FABs. The Kth element contains a FAB
91  that holds the data for the Kth grid, a Box that defines the valid region
92  of the Kth grid.
93 
94  A typical use for a FabArray would be to hold the solution vector or
95  right-hand-side when solving a linear system of equations on a union of
96  rectangular grids. The copy operations would be used to copy data from the
97  valid regions of neighboring grids into the boundary regions after each
98  relaxation step of the iterative method. If a multigrid method is used, a
99  FabArray could be used to hold the data at each level in the multigrid
100  hierarchy.
101 
102  This class is a concrete class not a polymorphic one.
103 
104  This class does NOT provide a copy constructor or assignment operator.
105 */

1443  const int N_rcvs = thecpc.m_RcvTags->size();
1444  const int N_locs = thecpc.m_LocTags->size();
1445 
1446  if (N_locs == 0 && N_rcvs == 0 && N_snds == 0)
1447  //
1448  // No work to do.
1449  //
1450  return;
1451 
1452 #ifdef BL_USE_MPI3
1453  MPI_Group tgroup, rgroup, sgroup;
1454  if (ParallelDescriptor::MPIOneSided()) {
1455  MPI_Comm_group(ParallelDescriptor::Communicator(this->color()), &tgroup);
1456  }
1457 #endif
1458 
1459  //
1460  // Send/Recv at most MaxComp components at a time to cut down memory usage.
1461  //
1462  int NCompLeft = ncomp;
1463 
1464  for (int ipass = 0, SC = scomp, DC = dcomp; ipass < ncomp; )
1465  {
1466  const int NC = std::min(NCompLeft,FabArrayBase::MaxComp);
1467 
1468  //
1469  // Before we post recv, let's preprocess sends in case FAB is not preAllocatable
1470  //
1471 
1472  Vector<char*> send_data;
1473  Vector<int> send_size;
1474  Vector<int> send_rank;
1475  Vector<MPI_Request> send_reqs;
1476  Vector<const CopyComTagsContainer*> send_cctc;
1477  Vector<Vector<int> > indv_send_size;
1478  Vector<MPI_Request> pre_reqs;
1479 
1480 #if defined BL_USE_UPCXX || defined BL_USE_MPI3
1481  int actual_n_snds = 0;
1482 #endif
1483  if (N_snds > 0)
1484  {
1485  send_data.reserve(N_snds);
1486  send_size.reserve(N_snds);
1487  send_rank.reserve(N_snds);
1488  send_reqs.reserve(N_snds);
1489  send_cctc.reserve(N_snds);
1490  indv_send_size.reserve(N_snds);
1491 
1492  for (auto const& kv : *thecpc.m_SndVols)
1493  {
1494  Vector<int> iss;
1495  auto const& cctc = thecpc.m_SndTags->at(kv.first);
1496 
1497  std::size_t nbytes = 0;
1498  if (FAB::preAllocatable())
1499  {
1500  for (auto const& cct : kv.second)
1501  {
1502  nbytes += src[cct.srcIndex].nBytes(cct.sbox,SC,NC);
1503  }
1504  }
1505  else
1506  {
1507  for (auto const& tag : cctc)
1508  {
1509  std::size_t b = src[tag.srcIndex].nBytes(tag.sbox,SC,NC);
1510  nbytes += b;
1511  iss.push_back(static_cast<int>(b));
1512  }
1513  }
1514 
1515  BL_ASSERT(nbytes < std::numeric_limits<int>::max());
1516 
1517  char* data = nullptr;
1518  if (nbytes > 0)
1519  {
1520  data = static_cast<char*>
1521 #ifdef BL_USE_UPCXX
1522  (BLPgas::alloc(nbytes));
1523 #else
1524  (amrex::The_Arena()->alloc(nbytes));
1525 #endif
1526  }
1527 
1528  send_data.push_back(data);
1529  send_size.push_back(static_cast<int>(nbytes));
1530  send_rank.push_back(kv.first);
1531  send_reqs.push_back(MPI_REQUEST_NULL);
1532  send_cctc.push_back(&cctc);
1533  indv_send_size.push_back(std::move(iss));
1534  }
1535 
1536 #if defined BL_USE_UPCXX || defined BL_USE_MPI3
1537  actual_n_snds = N_snds - std::count(send_size.begin(), send_size.end(), 0);
1538 #endif
1539  }
1540 
1541  if (!FAB::preAllocatable())
1542  {
1543  pre_reqs.resize(N_snds,MPI_REQUEST_NULL);
1544  for (int j=0; j<N_snds; ++j)
1545  {
1546  pre_reqs[j] = ParallelDescriptor::Asend(indv_send_size[j].data(),
1547  indv_send_size[j].size(),
1548  send_rank[j],preSeqNum).req();
1549  }
1550  }
1551 
1552  Vector<int> recv_from;
1553  Vector<char*> recv_data;
1554  Vector<int> recv_size;
1555  Vector<MPI_Request> recv_reqs;
1556 #ifdef BL_USE_MPI3
1557  Vector<MPI_Aint> recv_disp;
1558 #endif
1559  //
1560  // Post rcvs. Allocate one chunk of space to hold'm all.
1561  //
1562  char* the_recv_data = nullptr;
1563 
1564  int actual_n_rcvs = 0;
1565  if (N_rcvs > 0) {
1566 #ifdef BL_USE_UPCXX
1567  PostRcvs_PGAS(*thecpc.m_RcvVols, the_recv_data, recv_data,
1568  recv_size, recv_from,
1569  SC, NC, SeqNum, &BLPgas::cp_recv_event);
1570 #else
1571  if (ParallelDescriptor::MPIOneSided()) {
1572 #if defined(BL_USE_MPI3)
1573  PostRcvs_MPI_Onesided(*thecpc.m_RcvVols, the_recv_data, recv_data,
1574  recv_size, recv_from, recv_reqs, recv_disp,
1575  SC, NC, SeqNum, ParallelDescriptor::cp_win);
1576  MPI_Group_incl(tgroup, recv_from.size(), recv_from.dataPtr(), &rgroup);
1577  MPI_Win_post(rgroup, 0, ParallelDescriptor::cp_win);
1578 #endif
1579  } else {
1580  PostRcvs(*thecpc.m_RcvVols, *thecpc.m_RcvTags,
1581  recv_data, recv_size, recv_from, recv_reqs, SC, NC, SeqNum, preSeqNum);
1582  }
1583 #endif
1584  actual_n_rcvs = N_rcvs - std::count(recv_size.begin(), recv_size.end(), 0);
1585  }
1586 
1587  //
1588  // Post send's
1589  //
1590  if (N_snds > 0)
1591  {
1592 #ifdef _OPENMP
1593 #pragma omp parallel for if (FAB::isCopyOMPSafe())
1594 #endif
1595  for (int j=0; j<N_snds; ++j)
1596  {
1597  char* dptr = send_data[j];
1598  if (dptr != nullptr)
1599  {
1600  auto const& cctc = *send_cctc[j];
1601  for (auto const& tag : cctc)
1602  {
1603  const Box& bx = tag.sbox;
1604  auto n = src[tag.srcIndex].copyToMem(bx,SC,NC,dptr);
1605  dptr += n;
1606  }
1607  BL_ASSERT(dptr == send_data[j] + send_size[j]);
1608  }
1609  }
1610 
1611 #ifdef BL_USE_UPCXX
1612 
1613  BLPgas::cp_send_counter = 0;
1614 
1615  for (int i=0; i<N_snds; ++i) {
1616  if (send_size[i] > 0) {
1617  BLPgas::Send(upcxx::global_ptr<void>((void *)send_data[i], upcxx::myrank()),
1618  send_rank[i], send_size[i], SeqNum,
1619  &BLPgas::cp_send_event, &BLPgas::cp_send_counter);
1620  }
1621  }
1622 
1623  // Need to make sure at least half of the sends have been started
1624  while (BLPgas::cp_send_counter < actual_n_snds) {
1625  upcxx::advance();
1626  }
1627 
1628 #else
1629 
1630  if (ParallelDescriptor::MPIOneSided())
1631  {
1632 #if defined(BL_USE_MPI3)
1633  Vector<MPI_Aint> send_disp(N_snds,0);
1634  send_reqs.clear();
1635 
1636  for (int i=0; i<N_snds; ++i) {
1637  if (send_size[i] > 0) {
1638  send_reqs[i] = ParallelDescriptor::Arecv
1639  (&send_disp[i],1,send_rank[i],SeqNum).req();
1640  }
1641  }
1642 
1643  MPI_Group_incl(tgroup, N_snds, send_rank.data(), &sgroup);
1644  MPI_Win_start(sgroup,0,ParallelDescriptor::cp_win);
1645 
1646  int send_counter = 0;
1647  while (send_counter < actual_n_snds) {
1648  MPI_Status status;
1649  int index;
1650 
1651  MPI_Waitany(N_snds, send_reqs.dataPtr(), &index, &status);
1652 
1653  BL_ASSERT(status.MPI_TAG == SeqNum);
1654  BL_ASSERT(status.MPI_SOURCE == send_rank[index]);
1655 
1656  MPI_Put(send_data[index], send_size[index], MPI_CHAR, send_rank[index],
1657  send_disp[index], send_size[index], MPI_CHAR, ParallelDescriptor::cp_win);
1658 
1659  ++send_counter;
1660  }
1661 #endif
1662  }
1663  else
1664  {
1665  int send_counter = 0;
1666  while (send_counter < N_snds)
1667  {
1668  int j;
1669 
1670  if (FAB::preAllocatable())
1671  {
1672  j = send_counter;
1673  }
1674  else
1675  {
1676  MPI_Status status;
1677  MPI_Waitany(N_snds, pre_reqs.data(), &j, &status);
1678  }
1679 
1680  if (send_size[j] > 0)
1681  {
1682  if (FabArrayBase::do_async_sends)
1683  {
1684  send_reqs[j] = ParallelDescriptor::Asend
1685  (send_data[j],send_size[j],send_rank[j],SeqNum).req();
1686  }
1687  else
1688  {
1689  ParallelDescriptor::Send(send_data[j],send_size[j],send_rank[j],SeqNum);
1690  amrex::The_Arena()->free(send_data[j]);
1691  }
1692  }
1693 
1694  ++send_counter;
1695  }
1696  }
1697 #endif
1698  }
1699 
1700 #ifdef BL_USE_MPI3
1701  if (ParallelDescriptor::MPIOneSided()) {
1702  if (N_rcvs > 0) MPI_Group_free(&rgroup);
1703  if (N_snds > 0) MPI_Group_free(&sgroup);
1704  }
1705 #endif
1706 
1707  //
1708  // Do the local work. Hope for a bit of communication/computation overlap.
1709  //
1710  if (ParallelDescriptor::TeamSize() > 1 && thecpc.m_threadsafe_loc)
1711  {
1712 #ifdef BL_USE_TEAM
1713 #ifdef _OPENMP
1714 #pragma omp parallel if (FAB::isCopyOMPSafe())
1715 #endif
1716  ParallelDescriptor::team_for(0, N_locs, [&] (int j)
1717  {
1718  const CopyComTag& tag = (*thecpc.m_LocTags)[j];
1719 
1720  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
1721  // avoid self copy or plus
1722  if (op == FabArrayBase::COPY) {
1723  get(tag.dstIndex).copy(src[tag.srcIndex],tag.sbox,SC,tag.dbox,DC,NC);
1724  } else {
1725  get(tag.dstIndex).plus(src[tag.srcIndex],tag.sbox,tag.dbox,SC,DC,NC);
1726  }
1727  }
1728  });
1729 #endif
1730  }
1731  else
1732  {
1733 #ifdef _OPENMP
1734 #pragma omp parallel for if (FAB::isCopyOMPSafe() && thecpc.m_threadsafe_loc)
1735 #endif
1736  for (int j=0; j<N_locs; ++j)
1737  {
1738  const CopyComTag& tag = (*thecpc.m_LocTags)[j];
1739 
1740  if (this != &src || tag.dstIndex != tag.srcIndex || tag.sbox != tag.dbox) {
1741  // avoid self copy or plus
1742  if (op == FabArrayBase::COPY) {
1743  get(tag.dstIndex).copy(src[tag.srcIndex],tag.sbox,SC,tag.dbox,DC,NC);
1744  } else {
1745  get(tag.dstIndex).plus(src[tag.srcIndex],tag.sbox,tag.dbox,SC,DC,NC);
1746  }
1747  }
1748  }
1749  }
1750 
1751  //
1752  // wait and unpack
1753  //
1754 #ifdef BL_USE_UPCXX
1755  if (actual_n_rcvs > 0) BLPgas::cp_recv_event.wait();
1756 #else
1757  if (ParallelDescriptor::MPIOneSided()) {
1758 #if defined(BL_USE_MPI3)
1759  if (N_snds > 0) MPI_Win_complete(ParallelDescriptor::cp_win);
1760  if (N_rcvs > 0) MPI_Win_wait (ParallelDescriptor::cp_win);
1761 #endif
1762  } else {
1763  if (actual_n_rcvs > 0) {
1764  Vector<MPI_Status> stats(N_rcvs);
1765  BL_MPI_REQUIRE( MPI_Waitall(N_rcvs, recv_reqs.dataPtr(), stats.dataPtr()) );
1766  if (!CheckRcvStats(stats, recv_size, MPI_CHAR, SeqNum))
1767  {
1768  amrex::Abort("ParallelCopy failed with wrong message size");
1769  }
1770  }
1771  }
1772 #endif
1773 
1774  if (N_rcvs > 0)
1775  {
1776  Vector<const CopyComTagsContainer*> recv_cctc(N_rcvs,nullptr);
1777 
1778  for (int k = 0; k < N_rcvs; ++k)
1779  {
1780  if (recv_size[k] > 0)
1781  {
1782  auto const& cctc = thecpc.m_RcvTags->at(recv_from[k]);
1783  recv_cctc[k] = &cctc;
1784  }
1785  }
1786 
1787 #ifdef _OPENMP
1788 #pragma omp parallel if (FAB::isCopyOMPSafe() && thecpc.m_threadsafe_rcv)
1789 #endif
1790  {
1791  FAB fab;
1792 
1793 #ifdef _OPENMP
1794 #pragma omp for
1795 #endif
1796  for (int k = 0; k < N_rcvs; ++k)
1797  {
1798  const char* dptr = recv_data[k];
1799  if (dptr != nullptr)
1800  {
1801  auto const& cctc = *recv_cctc[k];
1802  for (auto const& tag : cctc)
1803  {
1804  const Box& bx = tag.dbox;
1805  std::size_t n;
1806  if (op == FabArrayBase::COPY)
1807  {
1808  n = get(tag.dstIndex).copyFromMem(bx,DC,NC,dptr);
1809  }
1810  else
1811  {
1812  fab.resize(bx,NC);
1813  n = fab.copyFromMem(bx,0,NC,dptr);
1814  get(tag.dstIndex).plus(fab,bx,bx,0,DC,NC);
1815  }
1816  dptr += n;
1817  }
1818  BL_ASSERT(dptr == recv_data[k] + recv_size[k]);
1819  }
1820  }
1821  }
1822 
1823  if (the_recv_data)
1824  {
1825 #ifdef BL_USE_UPCXX
1826  BLPgas::free(the_recv_data);
1827 #else
1828  if (ParallelDescriptor::MPIOneSided()) {
1829 #if defined(BL_USE_MPI3)
1830  MPI_Win_detach(ParallelDescriptor::cp_win, the_recv_data);
1831 #endif
1832  }
1833  amrex::The_Arena()->free(the_recv_data);
1834 #endif
1835  }
1836  else
1837  {
1838  for (auto p : recv_data) {
1839  amrex::The_Arena()->free(p);
1840  }
1841  }
1842  }
1843 
1844  if (N_snds > 0) {
1845 #ifdef BL_USE_UPCXX
1846  FabArrayBase::WaitForAsyncSends_PGAS(N_snds,send_data,
1847  &BLPgas::cp_send_event,
1848  &BLPgas::cp_send_counter);
1849 #else
1850  if (ParallelDescriptor::MPIOneSided()) {
1851 #if defined(BL_USE_MPI3)
1852  for (int i = 0; i < N_snds; ++i) {
1853  if (send_data[i]) amrex::The_Arena()->free(send_data[i]);
1854  }
1855 #endif
1856  } else {
1857  if (FabArrayBase::do_async_sends && ! thecpc.m_SndTags->empty()) {
1858  Vector<MPI_Status> stats;
1859  FabArrayBase::WaitForAsyncSends(N_snds,send_reqs,send_data,stats);
1860  }
1861  }
1862 #endif
1863  }
1864 
1865  ipass += NC;
1866  SC += NC;
1867  DC += NC;
1868  NCompLeft -= NC;
1869  }
1870 
1871 #ifdef BL_USE_MPI3
1872  if (ParallelDescriptor::MPIOneSided()) {
1873  MPI_Group_free(&tgroup);
1874  }
1875 #endif
1876 
1877 #ifdef BL_USE_TEAM
1878  ParallelDescriptor::MyTeam().MemoryBarrier();
1879 #endif
1880 
1881  return;
1882 
1883 #endif /*BL_USE_MPI*/
1884 }
